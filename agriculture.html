Problem Statement 2: Enhancing The Agricultural Facts

i) Crop Yield Optimization

Problem: Farmers need to identify the highest-yielding combinations to maximize overall production.

Solution: Sort data by yield (descending) and analyze to identify top-performing combinations. This helps farmers make informed decisions on crop selection and timing.

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct CropData {
    string field;
    string variety;
    string season;
    double yield;

    CropData(string f, string v, string s, double y)
        : field(f), variety(v), season(s), yield(y) {}
};

bool compareByYield(const CropData &a, const CropData &b) {
    return a.yield > b.yield;
}

void displayTopCombinations(const vector<CropData> &data) {
    cout << "Top-performing combinations (sorted by yield):\n";
    for (const auto &entry : data) {
        cout << "Field: " << entry.field
             << ", Variety: " << entry.variety
             << ", Season: " << entry.season
             << ", Yield: " << entry.yield << endl;
    }
}

int main() {
    vector<CropData> cropData = {
        CropData("Field1", "VarietyA", "Spring", 10.5),
        CropData("Field2", "VarietyB", "Summer", 15.3),
        CropData("Field1", "VarietyA", "Fall", 20.2),
        CropData("Field3", "VarietyC", "Spring", 18.7),
        CropData("Field2", "VarietyB", "Fall", 25.1),
        CropData("Field3", "VarietyC", "Summer", 22.4),
        CropData("Field1", "VarietyC", "Summer", 30.0)
    };

    sort(cropData.begin(), cropData.end(), compareByYield);
    displayTopCombinations(cropData);

    return 0;
}

ii) Irrigation Network Design

Problem: Designing efficient irrigation systems for optimal water distribution across fields.

Solution: Use graph modeling and algorithms like Dijkstraâ€™s to find the shortest paths for water flow, minimizing loss and maximizing coverage.

#include <iostream>
#include <vector>
#include <queue>
#include <climits>

using namespace std;

struct Edge {
    int to;
    int weight;
};

vector<int> dijkstra(int n, const vector<vector<Edge>>& graph, int source) {
    vector<int> dist(n, INT_MAX);
    dist[source] = 0;

    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, source});

    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();

        for (const auto& edge : graph[u]) {
            int v = edge.to;
            int weight = edge.weight;

            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                pq.push({dist[v], v});
            }
        }
    }

    return dist;
}

int main() {
    int n = 6;

    vector<vector<Edge>> graph(n);

    graph[0].push_back({1, 2});
    graph[0].push_back({2, 4});
    graph[1].push_back({3, 1});
    graph[1].push_back({2, 3});
    graph[2].push_back({4, 5});
    graph[3].push_back({5, 3});
    graph[4].push_back({5, 1});

    vector<int> dist = dijkstra(n, graph, 0);

    cout << "Shortest distances from the water source (Node 0):\n";
    for (int i = 0; i < n; i++) {
        if (dist[i] == INT_MAX) {
            cout << "Node " << i << ": No path\n";
        } else {
            cout << "Node " << i << ": " << dist[i] << " units of distance\n";
        }
    }

    return 0;
}
